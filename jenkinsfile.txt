pipeline {
    agent any
    stages {
        stage('Checkout code'){
            steps{
                checkout([$class: 'GitSCM', branches: [[name: '*/master']], doGenerateSubmoduleConfigurations: false, extensions: [], submoduleCfg: [], userRemoteConfigs: [[url: 'https://github.com/heenasharma11/java_maven_CICD_app']]])
            }
        }
        
        stage('verify mvn') {
            steps {
               sh 'mvn --version'
			   MVN_HOME = tool 'Maven'
            }
        }
        
        stage('Build') {
           steps {
				sh '"$MVN_HOME/bin/mvn" -Dmaven.test.failure.ignore clean package'
            }
        }
	stage('package') {
            steps {
                sh 'mvn -Dmaven.test.failure.ignore=true install'
            }
        }
    }
    
    post {
            success {
                echo 'this will run if successful' 
                }
    }      
}
